# Example

```python
from connections.google_analytics import GoogleAnalytics
import json


client = GoogleAnalytics(
    211168255,
    '/users/mage/mage-241122-6ca7fe21b556.json',
)

data = client.load(
    '2022-10-01',
    dimensions=[
        'city',
        'country',
    ],
    metrics=[
        'activeUsers',
        'newUsers',
        'sessions',
        'totalUsers',
    ],
)

print(json.dumps(data[:5], indent=2))
```
