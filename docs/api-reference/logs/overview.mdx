<ResponseField name="block_run_logs" type="array of objects" required>
  Logs for block runs.

  <Expandable title="properties" defaultOpen="true">
    <ResponseField name="name" type="string" required>
      File name of log file.
    </ResponseField>

    <ResponseField name="path" type="string" required>
      Full path of log file.
    </ResponseField>

    <ResponseField name="content" type="string" required>
      Contents of log file.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="pipeline_run_logs" type="array of objects" required>
  Logs for pipeline runs.

  <Expandable title="properties" defaultOpen="true">
    <ResponseField name="name" type="string" required>
      File name of log file.
    </ResponseField>

    <ResponseField name="path" type="string" required>
      Full path of log file.
    </ResponseField>

    <ResponseField name="content" type="string" required>
      Contents of log file.
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="total_block_run_log_count" type="integer" required>
  Number of block run logs in the API response.
</ResponseField>

<ResponseField name="total_pipeline_run_log_count" type="integer" required>
  Number of pipeline run logs in the API response.
</ResponseField>
